#### 带有编号的显示公式
\begin{equation}
    R^s_r(t_r,t_e)=(t_r-t_e)c
\end{equation}
***
#### 带有编号的多行公式
\begin{eqnarray}
    R^s_r(t_r,t_e)=(t_r-t_e)c \\\
    R^s_r(t_r,t_e)=(t_r-t_e)c \\\
    R^s_r(t_r,t_e)=(t_r-t_e)c
\end{eqnarray}
***
#### 取消某一编号的多行公式
\begin{eqnarray}
    R^s_r(t_r,t_e)=(t_r-t_e)c \\\
    R^s_r(t_r,t_e)=(t_r-t_e)c \nonumber\\\
    R^s_r(t_r,t_e)=(t_r-t_e)c
\end{eqnarray}
***
#### 带有指定编号的多行公式
\begin{eqnarray\*}
    R^s_r(t_r,t_e)=(t_r-t_e)c \\\
    R^s_r(t_r,t_e)=(t_r-t_e)c \tag{1}\\\
    R^s_r(t_r,t_e)=(t_r-t_e)c
\end{eqnarray\*}
***
#### 不带编号的多行公式
\begin{eqnarray\*}
    R^s_r(t_r,t_e)=(t_r-t_e)c \\\
    R^s_r(t_r,t_e)=(t_r-t_e)c \\\
    R^s_r(t_r,t_e)=(t_r-t_e)c
\end{eqnarray\*}
***
#### 指定对齐方式的多行公式
\begin{eqnarray}
    &R^s_r(t_r,t_e)&=(t_r-t_e)c \\\

\end{eqnarray}
